const noBtn = document.getElementById("noBtn");

const page1 = document.getElementById("page1");

  [yesSound, noSound, bgMusic].forEach(audio => {
    audio.play().then(() => audio.pause());
  });

// Floating hearts
for(let i=0;i<18;i++){
  const heart = document.createElement("div");
  heart.className="heart";
  heart.style.left=Math.random()*100+"vw";
  heart.style.animationDelay=Math.random()*8+"s";
  heart.style.animationDuration=6+Math.random()*6+"s";
  document.body.appendChild(heart);
}

function moveNo(){
  noSound.currentTime=0; noSound.play();
  const x = Math.random()*240;
  const y = Math.random()*160;
  noBtn.style.left=x+"px"; noBtn.style.top=y+"px";
}

noBtn.addEventListener("mouseenter", moveNo);
noBtn.addEventListener("click", moveNo);

yesBtn.addEventListener("click", ()=>{
  yesSound.currentTime=0; yesSound.play();
  launchConfetti();
  setTimeout(()=>{ page1.style.display="none"; page2.style.display="flex"; }, 1200);
});

function launchConfetti(){
  for(let i=0;i<200;i++){
    const confetti = document.createElement("div");
    confetti.className = "confetti";
    confetti.style.left = Math.random() * 100 + "vw";
    confetti.style.background = `hsl(${Math.random()*360},100%,60%)`;
    confetti.style.animationDuration = 2 + Math.random()*2 + "s";
    confetti.style.transform = `rotate(${Math.random()*360}deg)`;
    document.body.appendChild(confetti);

    setTimeout(()=>confetti.remove(),4000);
  }
}
